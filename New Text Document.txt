   <script>
         let plusBtn = document.getElementById('plus-btn');
         let minusBtn = document.getElementById('minus-btn');
         let inputFieldDisplay = document.getElementById('input-field');
         let addCaseBtn = document.getElementById('add-case');
         let removeCaseBtn = document.getElementById('remove-case');
         let inputField = document.getElementById('input-casing');
         let phoneQty = document.getElementById('phn-qunt');
         let itemQty = document.getElementById('item-qty');
         let subTotal = document.getElementById('sub-total');
         let totalAmount = document.getElementById('total-amount');
         let removePhone = document.getElementById('remove-phn');
         let removeCasing = document.getElementById('remove-casing'); 
         let addProduct = 0;
         let addCase = 0;
         let taxRate = 0.15;
         let phnPrice = 1219
         let casePrice = 59
         let btnCheckOut = document.getElementById('btn-checkout');
         let goBackBtn = document.getElementById('btn-go-back');
         document.getElementById('phn-price').innerText = phnPrice;
         document.getElementById('casing-price').innerText = casePrice;
         document.getElementById('tax-rate').innerText = taxRate *100;
         plusBtn.addEventListener('click', function() { 
            minusBtn.disabled = false;
            addProduct++;
            inputFieldDisplay.value = addProduct;
            phoneQty.innerText = addProduct;
            subTotal.innerText = parseInt(subTotal.innerText) + phnPrice;
            totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
         });
         minusBtn.addEventListener('click', function() { 
            if (addProduct === 0) {
               minusBtn.disabled = true;
               if (addCase === 0) {
                     totalAmount.innerText = 0;
               }
            } else {
               addProduct--;
               inputFieldDisplay.value = addProduct;
               phoneQty.innerText = addProduct;
               subTotal.innerText = parseInt(subTotal.innerText) - phnPrice;
               totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
            }
         });
         addCaseBtn.addEventListener('click', function() { 
            removeCaseBtn.disabled = false;
            addCase++;
            inputField.value = addCase;
            itemQty.innerText = addCase;
            subTotal.innerText = parseInt(subTotal.innerText) + casePrice;
            totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
         });
         removeCaseBtn.addEventListener('click', function() { 
            if (addCase === 0) {
               removeCaseBtn.disabled = true;
               if (addProduct === 0) {
                     totalAmount.innerText = 0;
               }
            } else {
               addCase--;
               inputField.value = addCase;
               itemQty.innerText = addCase;
               subTotal.innerText = parseInt(subTotal.innerText) - casePrice;
               totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
            }
         });
         removePhone.addEventListener('click', function() { 
            if (addCase > 0) {
               inputFieldDisplay.value = 0;
               addProduct = 0;
               phoneQty.innerText = 0;
               minusBtn.disabled = true;
               subTotal.innerText = addCase * casePrice ;
               totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
            } else {
               inputFieldDisplay.value = 0;
               addProduct = 0;
               phoneQty.innerText = 0;
               totalAmount.innerText = 0;
               subTotal.innerText = 0;
               minusBtn.disabled = true;
            }
         });
         removeCasing.addEventListener('click', function() { 
            if (addProduct > 0) {
               inputField.value = 0;
               addCase = 0;
               caseQty.innerText = 0;
               removeCaseBtn.disabled = true;
               subTotal.innerText = addProduct * phnPrice ;
               totalAmount.innerText = parseInt(subTotal.innerText) + (parseInt(subTotal.innerText)*taxRate);
            } else {
               inputField.value = 0;
               addCase = 0;
               itemQty.innerText = 0;
               totalAmount.innerText = 0;
               subTotal.innerText = 0;
               removeCaseBtn.disabled = true;
            }
         });
   </script>
